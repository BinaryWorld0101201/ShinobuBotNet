1. create users table:
CREATE TABLE `users` ( `id` INT(100) UNSIGNED NOT NULL AUTO_INCREMENT , `PC_user` VARCHAR(100) NOT NULL , `ip` VARCHAR(100) NOT NULL , `online` VARCHAR(100) NOT NULL , UNIQUE `id` (`id`)) ENGINE = InnoDB CHARSET=utf8 COLLATE utf8_general_ci;

2. create command table:
CREATE TABLE `command` ( `id_user` INT(100) NOT NULL AUTO_INCREMENT , `commandType` VARCHAR(100) NOT NULL , `commandContent` VARCHAR(100) NOT NULL , UNIQUE `id_user` (`id_user`)) ENGINE = InnoDB CHARSET=utf8 COLLATE utf8_general_ci;
